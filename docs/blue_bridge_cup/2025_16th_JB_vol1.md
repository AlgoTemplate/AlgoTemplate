#### 第六题

[数组翻转](https://www.lanqiao.cn/problems/20526/learning/)

??? example "3分"

    === "C++"
        ```cpp
        #include <bits/stdc++.h>
        using namespace std;
        typedef long long ll;
        const int N=1e6+10;

        int a[N],b[N],n;

        void reverse(int l,int r){
            while(l<r){
                int t=a[l];
                a[l]=a[r];
                a[r]=t;
                l++;
                r--;
            }
        }

        int main(){
            ios::sync_with_stdio(false);
            cin.tie(0);

            cin>>n;

            for(int i=1;i<=n;i++){
                cin>>a[i];
                b[i]=a[i];
            }

            ll ans=0;

            for(int l=1;l<=n;l++){
                for(int r=l;r<=n;r++){

                    reverse(l,r);

                    int pre=a[1],cnt=1;

                    for(int k=2;k<=n;k++){
                        if(a[k]==a[k-1]){
                            cnt++;
                        }else{
                            ans=max(ans,1ll*pre*cnt);
                            pre=a[k];
                            cnt=1;
                        }
                    }

                    ans=max(ans,1ll*pre*cnt);

                    for(int k=1;k<=n;k++) a[k]=b[k];
                }
            }

            cout<<ans;

            return 0;
        }
        ```
