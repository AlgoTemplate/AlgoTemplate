
#### 第一题

[逃离高塔](https://www.lanqiao.cn/problems/20560/learning/)

??? example "5分"

    === "C++"
        ```cpp
        #include<bits/stdc++.h>
        using namespace std;
        typedef long long ll;

        int main(){
            int ans=0;
            for(int i=1;i<=2025;i++){
                if(1ll*i*i*i%10==3) ans++;
            }
            cout<<ans;
            return 0;
        }
        ```

    === "Java"
        ```java
        import java.util.*;

        public class Main {

            public static void main(String[] args){
                int ans=0;
                for(int i=1;i<=2025;i++){

                }
                System.out.println(ans);
            }

        }
        ```

#### 第二题

[消失的蓝宝](https://www.lanqiao.cn/problems/20553/learning/)

??? example "5分"

    === "C++"
        ```cpp
        #include<iostream>
        using namespace std;
        typedef long long ll;

        int main(){
            ll n=20240413;
            for(ll i=1;;i++){
                ll N=i*n-9999;
                if((N+20240413)%20250412==0){
                    cout<<N;
                    break;
                }
            }
            return 0;
        }
        ```

    === "Java"
        ```java
        import java.util.*;

        public class Main {

            public static void main(String[] args){
                long n=20240413;
                for(long i=1;;i++){
                    long N=i*n-9999;
                    if((N+20240413)%20250412==0){
                        System.out.println(N);
                        break;
                    }
                }
            }

        }
        ```

    === "Python"
        ```python
        n=20240413
        i=1

        while True:
            N=i*n-9999
            if (N+20240413)%20250412==0:
                print(N)
                break
            i+=1
        ```

#### 第三题

[电池分组](https://www.lanqiao.cn/problems/20547/learning/)
??? example "10分"

    === "C++"
        ```cpp
        #include<bits/stdc++.h>
        using namespace std;
        const int N=1e3+10;
        int a[N],n;

        int main(){
            ios::sync_with_stdio(false);
            cin.tie(0);
            int T;
            cin>>T;
            while(T--){
                cin>>n;
                int sum=0;
                for(int i=1;i<=n;i++){
                    cin>>a[i];
                    sum^=a[i];
                }
                if(sum==0) cout<<"YES\n";
                else cout<<"NO\n";
            }
            return 0;
        }
        ```

    === "Java"
        ```java
        import java.util.*;

        public class Main {

            static final int N=1010;
            static int[] a=new int[N];

            public static void main(String[] args){
                Scanner sc=new Scanner(System.in);
                int T=sc.nextInt();
                while(T-->0){
                    int n=sc.nextInt();
                    int sum=0;
                    for(int i=1;i<=n;i++){
                        a[i]=sc.nextInt();
                        sum^=a[i];
                    }
                    if(sum==0) System.out.println("YES");
                    else System.out.println("NO");
                }
            }

        }
        ```

    === "Python"
        ```python
        import sys
        input=sys.stdin.readline

        T=int(input())
        while T:
            T-=1
            n=int(input())
            a=list(map(int,input().split()))
            s=0
            for x in a:
                s^=x
            if s==0:
                print("YES")
            else:
                print("NO")
        ```
