[寻找质数](https://www.lanqiao.cn/problems/20557/learning/)

??? example "5分"

    === "C++"
        ```cpp
        #include <bits/stdc++.h>
        using namespace std;
        typedef long long ll;

        bool isprime(int x){
            for(int i=2;i*i<=x;i++){
                if(x%i==0) return false;
            }
            return true;
        }

        int main(){
            int cnt=0;

            for(int i=2;;i++){
                if(isprime(i)) cnt++;
                if(cnt==2025){
                    cout<<i;
                    break;
                }
            }

            return 0;
        }
        ```

        ```cpp
        #include <bits/stdc++.h>
        using namespace std;
        typedef long long ll;

        const int N=1e3+10;

        int a[N],b[N],c[N];

        int calc(int x,int y,int z){
            vector<int> v;
            v.push_back(x);
            v.push_back(y);
            v.push_back(z);

            if(v[0]+1==v[1] && v[1]+1==v[2]) return 200;

            sort(v.begin(),v.end());

            if(v[0]==v[1] && v[1]==v[2]) return 200;
            if(v[0]+1==v[1] && v[1]+1==v[2]) return 100;
            if(v[0]==v[1] || v[1]==v[2]) return 100;

            return 0;
        }

        int main(){
            int n;
            cin>>n;

            for(int i=1;i<=n;i++) cin>>a[i];
            for(int i=1;i<=n;i++) cin>>b[i];
            for(int i=1;i<=n;i++) cin>>c[i];

            int m;
            cin>>m;

            int x=1,y=1,z=1;
            int ans=0;

            while(m--){
                int x1,x2,x3;
                cin>>x1>>x2>>x3;

                x=(x+x1)%n;
                y=(y+x2)%n;
                z=(z+x3)%n;

                if(x==0) x=n;
                if(y==0) y=n;
                if(z==0) z=n;

            return 0;
        }
        ```

#### 第六题（官方数据偏弱）

??? example "15分（不水 12 分）"

    === "C++"
        ```cpp
        #include <bits/stdc++.h>

        typedef long long ll;

        int main(){
            string s;
            cin>>s;

            int n=s.size();
            ll ans=0;

            for(int i=0;i<n;i++){
                int pre=-1;
                int cnt=0;

                for(int j=i;j<n;j++){
                    if(pre==-1){
                        ans++;
                        pre=s[j];
                        continue;
                    }

                    if(s[j]==pre || s[j]==pre+1){
                        ans++;
                        pre=s[j];
                        continue;
                    }

                    cnt++;
                    if(cnt==2){
                        break;
                    }

                    ans++;
                    pre=s[j];
                }
            }

            cout<<ans;

            return 0;
        }
        ```
