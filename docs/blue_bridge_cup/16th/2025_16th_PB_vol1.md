#### 第一题

[攻击次数](https://www.lanqiao.cn/problems/20564/learning/)

??? example "5分"

    === "C++"
        ```cpp
        #include<bits/stdc++.h>
        using namespace std;

        int main(){
            int ret=2025;
            for(int i=1;;i++){
                ret-=5;
                if(i%2==1){
                    ret-=15;
                }else{
                    ret-=2;
                }
                if(i%3==1){
                    ret-=2;
                }else if(i%3==2){
                    ret-=10;
                }else{
                    ret-=7;
                }
                if(ret<=0){
                    cout<<i<<"\n";
                    break;
                }
            }
            return 0;
        }
        ```
